create table Utilisateur
(
id_Utilisateur        integer        not null,
nom_Utilisateur       char(10)       not null,
Prénom                char(10)       null    ,
Localisation          char(20)       not null,
constraint pk_Utilisateur primary key(id_Utilisateur)
)
;

create table Plante 
(
id_Plante        integer         not null,
nom_Plante       char(10)        not null,
Taille_min       integer         not null,
Taille_max       integer         not null,
Couleur          char(10)        not null,
Effet bénéfique  char(20)        not null,
Localisation     char(20)        not null,
id_Utilisateur   integer         not null,
constraint pk_plante primary key(id_plante)
constraint fk_plante foreign key(id_utilisateur)
references Utilisateur(id_Utilisateur)
)
;

create table Dernier_entretien
(
id_Entretien           integer          not null,
Temperature_optimal    integer          not null,
Luminosité             integer          not null,
date                   date        	not null,
Humidité               integer          not null,
id_plante              integer          not mull,
constraint pk_Dernier_entretien primary key(id_Entretien),
constraint fk_Dernier_entretien foreign key(id_plante) refereces Plante(id_plante),
)
;
